<script>
const CLIENT_ID = '504528754299-ita27njppi7v091vc4c5mrm61ufmiuih.apps.googleusercontent.com';
const SCOPES = 'https://www.googleapis.com/auth/photoslibrary.readonly https://www.googleapis.com/auth/photoslibrary.appendonly';

let accessToken = null;
let photos = [];
let currentIndex = 0;
let selectedCat = null;
let stats = { obras: 0, familia: 0, clientes: 0, lixo: 0 };
let tokenClient = null;

// в”Ђв”Ђв”Ђ AUTH в”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђ
function initGoogleAuth() {
  tokenClient = google.accounts.oauth2.initTokenClient({
    client_id: CLIENT_ID,
    scope: SCOPES,
    callback: (resp) => {
      if (resp.error) { showToast('Erro ao autenticar: ' + resp.error, 'error'); return; }
      accessToken = resp.access_token;
      onAuthSuccess();
    }
  });
}

function handleLogin() {
  if (!tokenClient) { initGoogleAuth(); }
  tokenClient.requestAccessToken();
}

function handleLogout() {
  if (accessToken) google.accounts.oauth2.revoke(accessToken);
  accessToken = null;
  photos = [];
  currentIndex = 0;
  stats = { obras:0, familia:0, clientes:0, lixo:0 };
  document.getElementById('user-info').style.display = 'none';
  document.getElementById('btn-login').style.display = '';
  showScreen('welcome');
}

async function onAuthSuccess() {
  // Buscar perfil do usuГЎrio
  try {
    const r = await fetch('https://www.googleapis.com/oauth2/v3/userinfo', {
      headers: { Authorization: 'Bearer ' + accessToken }
    });
    const u = await r.json();
    document.getElementById('user-avatar').src = u.picture || '';
    document.getElementById('user-name').textContent = u.name || u.email;
    document.getElementById('user-info').style.display = 'flex';
    document.getElementById('btn-login').style.display = 'none';
  } catch(e) {}

  showScreen('loading');
  await loadPhotos();
}

// в”Ђв”Ђв”Ђ CARREGAR FOTOS в”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђ
async function loadPhotos() {
  setLoadingMsg('Buscando fotos do Google Fotos...');
  photos = [];
  let pageToken = null;

  try {
    do {
      const body = { pageSize: 100, filters: { mediaTypeFilter: { mediaTypes: ['PHOTO'] } } };
      if (pageToken) body.pageToken = pageToken;

      const r = await fetch('https://photoslibrary.googleapis.com/v1/mediaItems:search', {
        method: 'POST',
        headers: {
          Authorization: 'Bearer ' + accessToken,
          'Content-Type': 'application/json'
        },
        body: JSON.stringify(body)
      });
      const data = await r.json();

      if (data.error) {
        showToast('Erro da API: ' + data.error.message, 'error');
        showScreen('welcome');
        return;
      }

      if (data.mediaItems) photos.push(...data.mediaItems);
      pageToken = data.nextPageToken;
      setLoadingMsg(`Carregando fotos... ${photos.length} encontradas`);

      // Limitar a 500 por sessГЈo para performance
      if (photos.length >= 500) break;

    } while (pageToken);

    if (photos.length === 0) {
      showToast('Nenhuma foto encontrada.', 'error');
      showScreen('welcome');
      return;
    }

    showToast(`${photos.length} fotos carregadas!`, 'success');
    currentIndex = 0;
    showScreen('triage');
    loadCurrentPhoto();

  } catch(e) {
    showToast('Erro ao carregar fotos: ' + e.message, 'error');
    showScreen('welcome');
  }
}

// в”Ђв”Ђв”Ђ TRIAGEM в”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђ
function loadCurrentPhoto() {
  if (currentIndex >= photos.length) { showScreen('done'); return; }

  const photo = photos[currentIndex];
  const img = document.getElementById('current-photo');

  // URL base com tamanho
  img.src = photo.baseUrl + '=w1200-h900';
  img.onerror = () => { img.src = photo.baseUrl; };

  // Data
  const dt = photo.mediaMetadata?.creationTime;
  document.getElementById('photo-date').textContent = dt
    ? new Date(dt).toLocaleDateString('pt-BR', { day:'2-digit', month:'short', year:'numeric' })
    : 'Data desconhecida';

  // Contador
  document.getElementById('photo-counter').textContent = `${currentIndex + 1} / ${photos.length}`;

  // Progresso
  const pct = Math.round((currentIndex / photos.length) * 100);
  document.getElementById('prog-bar').style.width = pct + '%';
  document.getElementById('prog-pct').textContent = pct + '%';
  document.getElementById('prog-label').textContent = `${currentIndex} triadas`;

  // Reset seleГ§ГЈo
  selectedCat = null;
  document.querySelectorAll('.cat-btn').forEach(b => b.classList.remove('active'));
  document.getElementById('btn-confirm').disabled = true;
  document.getElementById('obra-name-input').style.display = 'none';

  // SugestГЈo IA automГЎtica
  autoSuggest(photo);
}

function autoSuggest(photo) {
  document.getElementById('ai-cat-label').textContent = 'Analisando...';
  document.getElementById('ai-reason').textContent = 'Verificando metadados';

  // AnГЎlise por metadados (heurГ­stica simples enquanto nГЈo hГЎ visГЈo de mГЎquina)
  const filename = (photo.filename || '').toLowerCase();
  const desc = (photo.description || '').toLowerCase();
  const ts = photo.mediaMetadata?.creationTime;
  const hour = ts ? new Date(ts).getHours() : -1;

  let cat = 'obras';
  let reason = 'Foto sem contexto identificГЎvel вЂ” classificada como obra por padrГЈo.';

  if (/selfie|pessoa|familia|aniver|natal|fГ©rias|praia|viagem/.test(filename + desc)) {
    cat = 'familia'; reason = 'Nome ou descriГ§ГЈo sugere foto pessoal.';
  } else if (/cliente|reuniГЈo|visita|contrato/.test(filename + desc)) {
    cat = 'clientes'; reason = 'Parece relacionada a clientes.';

const albumCache = {};

async function getOrCreateAlbum(name) {
  if (albumCache[name]) return albumCache[name];

  // Listar ГЎlbuns existentes
  const r = await fetch('https://photoslibrary.googleapis.com/v1/albums?pageSize=50', {
    headers: { Authorization: 'Bearer ' + accessToken }
  });
  const data = await r.json();
  const existing = (data.albums || []).find(a => a.title === name);
  if (existing) { albumCache[name] = existing.id; return existing.id; }

  // Criar ГЎlbum
  const rc = await fetch('https://photoslibrary.googleapis.com/v1/albums', {
    method: 'POST',
    headers: { Authorization: 'Bearer ' + accessToken, 'Content-Type': 'application/json' },
    body: JSON.stringify({ album: { title: name } })
  });
  const album = await rc.json();
  albumCache[name] = album.id;
  return album.id;
}

async function addToAlbum(mediaItemId, albumName) {
  const albumId = await getOrCreateAlbum(albumName);
  await fetch(`https://photoslibrary.googleapis.com/v1/albums/${albumId}:batchAddMediaItems`, {
    method: 'POST',
    headers: { Authorization: 'Bearer ' + accessToken, 'Content-Type': 'application/json' },
    body: JSON.stringify({ mediaItemIds: [mediaItemId] })
  });
}

async function onAuthSuccess() {
  // Buscar perfil do usuГЎrio
  try {
    const r = await fetch('https://www.googleapis.com/oauth2/v3/userinfo', {
      headers: { Authorization: 'Bearer ' + accessToken }
    });
    const u = await r.json();
    document.getElementById('user-avatar').src = u.picture || '';
    document.getElementById('user-name').textContent = u.name || u.email;

  await loadPhotos();

// в”Ђв”Ђв”Ђ CARREGAR FOTOS в”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђ
async function loadPhotos() {
  setLoadingMsg('Buscando fotos do Google Fotos...');
  photos = [];
  let pageToken = null;

  try {
    do {
      const body = { pageSize: 100, filters: { mediaTypeFilter: { mediaTypes: ['PHOTO'] } } };
      if (pageToken) body.pageToken = pageToken;

      const r = await fetch('https://photoslibrary.googleapis.com/v1/mediaItems:search', {
        method: 'POST',
        headers: {
          Authorization: 'Bearer ' + accessToken,
          'Content-Type': 'application/json'
        },
        body: JSON.stringify(body)
      });
      const data = await r.json();

      if (data.error) {
        showToast('Erro da API: ' + data.error.message, 'error');
        showScreen('welcome');
        return;
      }

      if (data.mediaItems) photos.push(...data.mediaItems);
      pageToken = data.nextPageToken;
      setLoadingMsg(`Carregando fotos... ${photos.length} encontradas`);

      // Limitar a 500 por sessГЈo para performance
      if (photos.length >= 500) break;

    } while (pageToken);

    if (photos.length === 0) {
      showToast('Nenhuma foto encontrada.', 'error');
      showScreen('welcome');
      return;
    }

    showToast(`${photos.length} fotos carregadas!`, 'success');
    currentIndex = 0;
    showScreen('triage');
    loadCurrentPhoto();

  } catch(e) {
    showToast('Erro ao carregar fotos: ' + e.message, 'error');
    showScreen('welcome');
  }
}

// в”Ђв”Ђв”Ђ TRIAGEM в”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђ
function loadCurrentPhoto() {
  if (currentIndex >= photos.length) { showScreen('done'); return; }

  const photo = photos[currentIndex];
  const img = document.getElementById('current-photo');

  // URL base com tamanho
  img.src = photo.baseUrl + '=w1200-h900';
  img.onerror = () => { img.src = photo.baseUrl; };

  // Data
  const dt = photo.mediaMetadata?.creationTime;
  document.getElementById('photo-date').textContent = dt
    ? new Date(dt).toLocaleDateString('pt-BR', { day:'2-digit', month:'short', year:'numeric' })
    : 'Data desconhecida';

  // Contador
  document.getElementById('photo-counter').textContent = `${currentIndex + 1} / ${photos.length}`;

  // Progresso
  const pct = Math.round((currentIndex / photos.length) * 100);
  document.getElementById('prog-bar').style.width = pct + '%';
  document.getElementById('prog-pct').textContent = pct + '%';
  document.getElementById('prog-label').textContent = `${currentIndex} triadas`;

  // Reset seleГ§ГЈo
  selectedCat = null;
  document.querySelectorAll('.cat-btn').forEach(b => b.classList.remove('active'));
  document.getElementById('btn-confirm').disabled = true;
  document.getElementById('obra-name-input').style.display = 'none';

  // SugestГЈo IA automГЎtica
  autoSuggest(photo);
}

function autoSuggest(photo) {
  document.getElementById('ai-cat-label').textContent = 'Analisando...';
  document.getElementById('ai-reason').textContent = 'Verificando metadados';

  // AnГЎlise por metadados (heurГ­stica simples enquanto nГЈo hГЎ visГЈo de mГЎquina)
  const filename = (photo.filename || '').toLowerCase();
  const desc = (photo.description || '').toLowerCase();
  const ts = photo.mediaMetadata?.creationTime;
  const hour = ts ? new Date(ts).getHours() : -1;

  let cat = 'obras';
  let reason = 'Foto sem contexto identificГЎvel вЂ” classificada como obra por padrГЈo.';

  if (/selfie|pessoa|familia|aniver|natal|fГ©rias|praia|viagem/.test(filename + desc)) {
    cat = 'familia'; reason = 'Nome ou descriГ§ГЈo sugere foto pessoal.';
  } else if (/cliente|reuniГЈo|visita|contrato/.test(filename + desc)) {
    cat = 'clientes'; reason = 'Parece relacionada a clientes.';
  } else if (/screenshot|screen|meme|print|forward/.test(filename)) {
    cat = 'lixo'; reason = 'Screenshot ou imagem encaminhada.';
  } else if (filename.startsWith('img_') || filename.startsWith('photo_') || filename.startsWith('dsc')) {
    cat = 'obras'; reason = 'PadrГЈo tГ­pico de foto de campo.';
  }

  const labels = { obras:'рџЏ—пёЏ Obra', familia:'рџ‘ЁвЂЌрџ‘©вЂЌрџ‘§ FamГ­lia/Pessoal', clientes:'рџ‘Ґ Clientes', lixo:'рџ—‘пёЏ Lixo' };
  document.getElementById('ai-cat-label').textContent = labels[cat];
  document.getElementById('ai-reason').textContent = reason;

  selectCat(cat);
}

// Cache de ГЎlbuns criados nesta sessГЈo
const albumCache = {};

async function getOrCreateAlbum(name) {
  if (albumCache[name]) return albumCache[name];

  // Listar ГЎlbuns existentes
  const r = await fetch('https://photoslibrary.googleapis.com/v1/albums?pageSize=50', {
    headers: { Authorization: 'Bearer ' + accessToken }
  });
  const data = await r.json();
  const existing = (data.albums || []).find(a => a.title === name);
  if (existing) { albumCache[name] = existing.id; return existing.id; }

  // Criar ГЎlbum
  const rc = await fetch('https://photoslibrary.googleapis.com/v1/albums', {
    method: 'POST',
    headers: { Authorization: 'Bearer ' + accessToken, 'Content-Type': 'application/json' },
    body: JSON.stringify({ album: { title: name } })
  });
  const album = await rc.json();
  albumCache[name] = album.id;
  return album.id;
}

async function addToAlbum(mediaItemId, albumName) {
  const albumId = await getOrCreateAlbum(albumName);
  await fetch(`https://photoslibrary.googleapis.com/v1/albums/${albumId}:batchAddMediaItems`, {
    method: 'POST',
    headers: { Authorization: 'Bearer ' + accessToken, 'Content-Type': 'application/json' },
    body: JSON.stringify({ mediaItemIds: [mediaItemId] })
  });
}

async function loadPhotos() {
  setLoadingMsg('Buscando fotos do Google Fotos...');
  photos = [];
  let pageToken = null;

  try {
    do {
      const body = { pageSize: 100, filters: { mediaTypeFilter: { mediaTypes: ['PHOTO'] } } };
      if (pageToken) body.pageToken = pageToken;

      const r = await fetch('https://photoslibrary.googleapis.com/v1/mediaItems:search', {
        method: 'POST',
        headers: {
          Authorization: 'Bearer ' + accessToken,
          'Content-Type': 'application/json'
        },
        body: JSON.stringify(body)
      });
      const data = await r.json();

      if (data.error) {
        showToast('Erro da API: ' + data.error.message, 'error');
        showScreen('welcome');
        return;
      }

      if (data.mediaItems) photos.push(...data.mediaItems);
      pageToken = data.nextPageToken;
      setLoadingMsg(`Carregando fotos... ${photos.length} encontradas`);

      // Limitar a 500 por sessГЈo para performance
      if (photos.length >= 500) break;

// Cache de ГЎlbuns criados nesta sessГЈo
const albumCache = {};

async function getOrCreateAlbum(name) {
  if (albumCache[name]) return albumCache[name];

  // Listar ГЎlbuns existentes
  const r = await fetch('https://photoslibrary.googleapis.com/v1/albums?pageSize=50', {
    headers: { Authorization: 'Bearer ' + accessToken }
  });
  const data = await r.json();
  const existing = (data.albums || []).find(a => a.title === name);
  if (existing) { albumCache[name] = existing.id; return existing.id; }

  // Criar ГЎlbum
  const rc = await fetch('https://photoslibrary.googleapis.com/v1/albums', {
    method: 'POST',
    headers: { Authorization: 'Bearer ' + accessToken, 'Content-Type': 'application/json' },
    body: JSON.stringify({ album: { title: name } })
  });
  const album = await rc.json();
  albumCache[name] = album.id;
  return album.id;
}

async function addToAlbum(mediaItemId, albumName) {
  const albumId = await getOrCreateAlbum(albumName);
  await fetch(`https://photoslibrary.googleapis.com/v1/albums/${albumId}:batchAddMediaItems`, {
    method: 'POST',
    headers: { Authorization: 'Bearer ' + accessToken, 'Content-Type': 'application/json' },
    body: JSON.stringify({ mediaItemIds: [mediaItemId] })
  });
}
